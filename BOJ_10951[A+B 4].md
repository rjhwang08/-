```c
#include<stdio.h>

int main() {
	int A, B;
	while (scanf("%d %d", &A, &B) != EOF) {   // EOF 대신 2 사용가능
		printf("%d\n", A + B);
	}
}
```

```java
import java.util.*;
public class Main{
    public static void main(String args[]){
        Scanner sc = new Scanner(System.in);
        int a, b;
        while (sc.hasNextInt()) {
            a = sc.nextInt();
            b = sc.nextInt();
            System.out.println(a + b);
        }
    }
}
```
