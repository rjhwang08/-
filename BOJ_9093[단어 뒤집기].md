```c++
#include<stdio.h>
#include<iostream>
#include<string>
#include<stack>
using namespace std;


int main() {
	int T;
	cin >> T; 
	cin.ignore();    //getline()함수에 개행 문자가 들어가는 것을 방지하기 위함
	while (T--) {
		stack<char> stk;
		string str;
		//cin >> s;   <- cin은 공백문자가 있으면 입력을 끊음
		getline(cin, str);
		str += '\n';
		int i;
		for (i = 0; i < str.size(); i++) {
			if (str[i] == ' ' || str[i] == '\n') {
				while (!stk.empty()) {
					cout << stk.top();
					stk.pop();
				}
				cout << str[i];
			} else stk.push(str[i]);
		}
	}
	return 0;
}
```
