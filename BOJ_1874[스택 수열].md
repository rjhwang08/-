```c++
#include<stdio.h>
#include<iostream>
#include<string>
#include<stack>
using namespace std;


int main() {
	stack<int> stk;
	string ans = "YES";
	string str;
	int T;
	int A[100000];
	int cnt = 1;

	cin >> T;
	for (int i = 0; i < T; i++)  {
		cin >> A[i];
		if (i == 0 || cnt <= A[i]) {
			while (cnt <= A[i]) {
				stk.push(cnt);
				str.push_back('+');
				cnt++;
			}
		}

		if (stk.top() == A[i]) {
			stk.pop();
			str.push_back('-');
		}
		else {
			ans = "NO";
		}
	}
	if (ans == "YES") {
		for (char ch : str) {
			cout << ch << '\n';;
		}
	}
	else {
		cout << ans << '\n';
	}

	return 0;
}
```
