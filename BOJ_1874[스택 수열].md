```c++
// 내 정답
#include<stdio.h>
#include<iostream>
#include<string>
#include<stack>
using namespace std;


int main() {
	stack<int> stk;
	string ans = "YES";
	string str;
	int T;
	int A[100000];
	int cnt = 1;

	cin >> T;
	for (int i = 0; i < T; i++)  {
		cin >> A[i];
		if (i == 0 || cnt <= A[i]) {
			while (cnt <= A[i]) {
				stk.push(cnt);
				str.push_back('+');
				cnt++;
			}
		}

		if (stk.top() == A[i]) {
			stk.pop();
			str.push_back('-');
		}
		else {
			ans = "NO";
		}
	}
	if (ans == "YES") {
		for (char ch : str) {
			cout << ch << '\n';;
		}
	}
	else {
		cout << ans << '\n';
	}

	return 0;
}
```

```c++
// BOJ c++
#include <iostream>
#include <stack>
#include <string>
using namespace std;
int main() {
    stack<int> s;
    int n;
    string ans;

    cin >> n;

    int m = 0;

    while (n--) {
        int x;
        cin >> x;
        if (x > m) {
            while (x > m) {
                s.push(++m);
                ans += '+';
            }
            s.pop();
            ans += '-';
        } else {
            bool found = false;
            if (!s.empty()) {
                int top = s.top();
                s.pop();
                ans += '-';
                if (x == top) {
                    found = true;
                }
            }
            if (!found) {
                cout << "NO" << '\n';
                return 0;
            }
        }
    }
    for (auto x : ans) {
        cout << x << '\n';
    }
    return 0;
}
```

```java
// BOJ java
import java.util.*;

public class Main {
    public static void solve(int[] a) {
        int n = a.length;
        Stack<Integer> stack = new Stack<Integer>();
        int m = 0;
        StringBuilder sb = new StringBuilder();
        for(int x : a) {
            if (x > m) {
                while (x > m) {
                    stack.push(++m);
                    sb.append("+\n");
                }
                stack.pop();
                sb.append("-\n");
            } else {
                if (stack.peek() != x) {
                    System.out.println("NO");
                    return;
                }
                stack.pop();
                sb.append("-\n");
            }
        }
        System.out.println(sb);
    }
    public static void main(String args[]) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] a = new int[n];
        for (int i=0; i<n; i++) {
            a[i] = sc.nextInt();
        }
        solve(a);
    }
}
```
