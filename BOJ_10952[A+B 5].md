# 풀이1
```c++
#include<stdio.h>

int main() {
	int A, B;
	while (true) {
		scanf("%d %d", &A, &B);
		if (A == 0 && B == 0) break;
		printf("%d\n", A + B);
	}
}
```

```c++
#include<stdio.h>

int main() {
	int A, B;
	while (true) {
		scanf("%d %d", &A, &B);
		if ((A || B) == 0) break;
		printf("%d\n", A + B);
	}
}
```
