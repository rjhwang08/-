1. Java(오답 : i번째까지의 최대값 D[i] = D[i-1] + A[i]) -> 연속해서 마시는 경우를 고려하지 못함
```java
import java.io.*;

public class Main {	
	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		
		int n = Integer.parseInt(br.readLine());
		int[] A = new int[n+1];
		for(int i=1; i <= n; i++)
			A[i] = Integer.parseInt(br.readLine());
		
		int[] D = new int[n+1];
		int[] idx = new int[n+1];
		for(int i=1; i <= n; i++) {
			D[i] = A[i];
			int max = 0;
			idx[1] = -1;
			for(int j=1; j < i; j++) {
				if(max < D[j]) {
					if(j >= 2 && j == i-1 && idx[j] == j-1)
						continue;
					max = D[j];
					idx[i] = j;
				}
			}
			D[i] += max;
		}
		
		int ans = D[1];
		for(int i=2; i <= n; i++) {
			if(D[i] > ans)
				ans = D[i];
		}
		System.out.println(ans);
		br.close();
	}
}
```

```java
import java.io.*;

public class Main {	
	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		
		int n = Integer.parseInt(br.readLine());
		int[] A = new int[n+1];
		for(int i=1; i <= n; i++)
			A[i] = Integer.parseInt(br.readLine());
		
		int[][] D = new int[n+1][3];
		D[1][0] = 0;
		D[1][1] = A[1];
		D[1][2] = A[1];
		for(int i=2; i <= n; i++) {
			D[i][0] = Math.max(D[i-1][2], Math.max(D[i-1][1], D[i-1][0]));
			D[i][1] = D[i-1][0] + A[i];
			D[i][2] = D[i-1][1] + A[i];
		}

		System.out.println(Math.max(D[n][2], Math.max(D[n][1], D[n][0])));
		br.close();
	}
}
```
